Create Table dbo.BitcoinVal
(   
    bCoinClose float NOT NULL,
    dateB date NOT NULL,
    timeB time NOT NULL,
    dateTimeB varchar(255) NOT NULL,    
    Primary Key (dateTimeB),
    dateTimeS varchar(255) FOREIGN KEY REFERENCES dbo.SentimentValues(dateTimeS) 
);
